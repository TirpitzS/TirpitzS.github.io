list.c

```
#include <stdio.h>
#include "list.h"

#define TRUE (1)
#define FALSE (0)

static int IsEmpty(Node_t *header)
{
	assert((header == NULL));
	if (header->next == NULL)
		return TRUE;
	else
		return FALSE;
}

static Node_t *FindLast(Node_t *header)
{
	Node_t *tmp = header;
	assert((header == NULL));
	
	while ((NULL != tmp->next))
	{
		tmp = tmp->next;
	}
	
	return tmp;
}

Node_t *CreateList()
{
	Node_t *header = (Node_t *)malloc(sizeof(node_t));
	assert((header == NULL));
	header->next = NULL;
	
	return header;
}

Node_t *DelList(Node_t *header)
{
	Node_t *N1Node = header;
	Node_t *N2Node;
	assert((header == NULL));
	
	do
	{
		N2Node = N1Node;
		N1Node = N1Node->next;
		free(N2Node);
	}
	while((NULL != N1Node->next));
	
	free(N1Node);
	
	header = NULL;
}

int Append(int data, Node_t *header)
{
	assert((header == NULL));
	Node_t *tmp = (Node_t *)malloc(sizeof(Node_t));
	
	Node_t *t2 = FindLast();
	
	tmp->data = data;
	t2 ->next = tmp;
	
	return 0;
}

void PrintList(Node_t *header)
{
	assert((header->next == NULL));
	Node_t *tmp = header;
	
	while ((tmp->next != NULL))
	{
		printf("%d\n", tmp->data);
	}
}

int main()
{
	Node_t *header = CreateList();
	for(i=0;i<100;i++)
	{
		Append(i, header);
	}
	PrintList();
	
	return 0;
}
```
